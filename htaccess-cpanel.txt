RewriteEngine On

# Serve viagro.uk from /viagro subfolder
RewriteCond %{HTTP_HOST} ^(www\.)?viagro\.uk$ [NC]
RewriteCond %{REQUEST_URI} !^/viagro/
RewriteRule ^(.*)$ /viagro/$1 [L,QSA]

# ─── Enable Gzip / Deflate compression ───────────────────────────────────────
<IfModule mod_deflate.c>
  AddOutputFilterByType DEFLATE text/html text/css text/javascript application/javascript application/json image/svg+xml
</IfModule>

# ─── Serve WebP if browser supports it and .webp file exists ─────────────────
<IfModule mod_rewrite.c>
  RewriteCond %{HTTP_ACCEPT} image/webp
  RewriteCond %{REQUEST_FILENAME} \.(jpe?g|png)$
  RewriteCond %{REQUEST_FILENAME:R-map}.webp -f
  RewriteRule ^(.+)\.(jpe?g|png)$ $1.webp [T=image/webp,L]
</IfModule>

# ─── Browser caching ─────────────────────────────────────────────────────────
<IfModule mod_expires.c>
  ExpiresActive On

  # HTML — short cache, always revalidate
  ExpiresByType text/html                    "access plus 1 hour"

  # Versioned JS/CSS (Next.js adds content hashes to filenames)
  ExpiresByType application/javascript       "access plus 1 year"
  ExpiresByType text/css                     "access plus 1 year"

  # Images — long cache
  ExpiresByType image/webp                   "access plus 1 year"
  ExpiresByType image/jpeg                   "access plus 1 year"
  ExpiresByType image/png                    "access plus 1 year"
  ExpiresByType image/svg+xml                "access plus 1 year"
  ExpiresByType image/x-icon                 "access plus 1 year"

  # Fonts
  ExpiresByType font/woff2                   "access plus 1 year"
  ExpiresByType font/woff                    "access plus 1 year"
</IfModule>

# ─── Cache-Control headers ───────────────────────────────────────────────────
<IfModule mod_headers.c>
  # Versioned static assets (JS, CSS, images, fonts)
  <FilesMatch "\.(js|css|webp|jpg|jpeg|png|svg|woff2|woff)$">
    Header set Cache-Control "public, max-age=31536000, immutable"
  </FilesMatch>
  # HTML pages
  <FilesMatch "\.html$">
    Header set Cache-Control "public, max-age=3600, must-revalidate"
  </FilesMatch>
</IfModule>
